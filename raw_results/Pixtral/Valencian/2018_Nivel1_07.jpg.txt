   \[
   \text{Costo de 3 juegos en 2 juegos} = 8 \text{ euros} \implies \text{Costo de 1 juego en 2 juegos} = \frac{8}{3} \text{ euros}
   \]
   \[
   \text{Costo de 3 juegos en 5 juegos} = 11 \text{ euros} \implies \text{Costo de 1 juego en 5 juegos} = \frac{11}{3} \text{ euros}
   \]

   \[
   \text{Costo de 1 juego} = \frac{\frac{8}{3} + \frac{11}{3}}{2} = \frac{19}{6} \text{ euros} \approx 3.1667 \text{ euros}
