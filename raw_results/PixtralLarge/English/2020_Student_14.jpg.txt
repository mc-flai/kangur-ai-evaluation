\[ (a - \frac{1}{a})^2 = 0 \]

\[ a - \frac{1}{a} = 0 \]
\[ b - \frac{1}{b} = 0 \]
\[ c - \frac{1}{c} = 0 \]

\[ a = \frac{1}{a} \]
\[ b = \frac{1}{b} \]
\[ c = \frac{1}{c} \]
