\[ \text{Área de la parte superpuesta} = \frac{2016 \, \text{cm}^2}{4} = 504 \, \text{cm}^2 \]

\[ \text{Área total de la parte no superpuesta} = 2016 \, \text{cm}^2 - 504 \, \text{cm}^2 = 1512 \, \text{cm}^2 \]
